  ```elixir
  defmodule LatticeStripe.Testing.TestClock.Owner do
    @moduledoc false
    # Per-test cleanup GenServer. Tracks clock ids created during a test;
    # cleaned up on `on_exit` via `cleanup/2`.
    #
    # NOT `start_supervised!` — the Owner must outlive the test pid so
    # `on_exit` can call `delete/3` even when the test crashes or raises.
    # This mirrors `Ecto.Adapters.SQL.Sandbox.start_owner!/2`.

    use GenServer

    @spec start_owner!(keyword()) :: pid()
    def start_owner!(opts \\ []) do
      case GenServer.start_link(__MODULE__, opts) do
        {:ok, pid} -> pid
        {:error, reason} -> raise "Failed to start TestClock Owner: #{inspect(reason)}"
      end
    end

    @spec register(pid(), String.t()) :: :ok
    def register(owner, clock_id) when is_pid(owner) and is_binary(clock_id) do
      GenServer.call(owner, {:register, clock_id})
    end

    @spec registered(pid()) :: [String.t()]
    def registered(owner) when is_pid(owner) do
      GenServer.call(owner, :registered)
    end

    @spec cleanup(pid(), LatticeStripe.Client.t()) :: :ok
    def cleanup(owner, client) when is_pid(owner) do
      ids = registered(owner)

      # Best-effort delete — swallow errors. The Mix task is the backstop.
      # Check Finch pool health before calling delete/3 (Pitfall 5): if the
      # pool is already down, log and skip.
      if client_transport_alive?(client) do
        for id <- ids do
          try do
            LatticeStripe.TestHelpers.TestClock.delete(client, id)
          rescue
            _ -> :ok
          catch
            _, _ -> :ok
          end
        end
      end

      GenServer.stop(owner)
      :ok
    end

    defp client_transport_alive?(_client) do
      # TODO: inspect client.finch pool health. For v1 just return true —
      # the rescue/catch above handles any transport errors. Revisit if
      # Pitfall 5 bites in practice.
      true
    end

    # GenServer callbacks

    @impl true
    def init(_opts) do
      {:ok, %{clock_ids: []}}
    end

    @impl true
    def handle_call({:register, id}, _from, state) do
      {:reply, :ok, %{state | clock_ids: [id | state.clock_ids]}}
    end

    @impl true
    def handle_call(:registered, _from, state) do
      # Return in registration order (reverse so first-registered is first)
      {:reply, Enum.reverse(state.clock_ids), state}
    end
  end
  ```

  ```elixir
  defmodule LatticeStripe.Testing.TestClock do
    @moduledoc """
    ExUnit ergonomics for Stripe Test Clocks.

    This module provides a `use`-macro that users opt into from their own
    `ExUnit.CaseTemplate`. It wraps `LatticeStripe.TestHelpers.TestClock`
    with automatic per-test cleanup (via a lightweight GenServer Owner),
    automatic customer↔clock linkage (closing the silent-correctness
    footgun of forgetting `test_clock:` on `Customer.create/2`), and a
    human-friendly `advance/2` that takes time units.

    ## Usage

        defmodule MyApp.StripeCase do
          use ExUnit.CaseTemplate

          using do
            quote do
              use LatticeStripe.Testing.TestClock, client: MyApp.StripeClient
            end
          end
        end

        defmodule MyApp.BillingTest do
          use MyApp.StripeCase, async: true
          setup :with_test_clock

          test "sub renews after 30 days", %{test_clock: clock} do
            customer = create_customer(clock, email: "a@b.c")
            {:ok, sub} = MyApp.Billing.subscribe(customer, "price_monthly")
            advance(clock, days: 30)
            assert {:ok, %{status: "active"}} = MyApp.Billing.get_subscription(sub.id)
          end
        end

    ## Cleanup strategy

    Every clock created via `test_clock/1` is registered with a per-test
    `LatticeStripe.Testing.TestClock.Owner` GenServer. On test exit
    (including crash / assertion failure), the Owner deletes each
    registered clock via `LatticeStripe.TestHelpers.TestClock.delete/3`.
    Stripe's delete cascades to attached Customers and Subscriptions.

    For SIGKILL / BEAM crash / CI timeout scenarios that bypass `on_exit`,
    the `mix lattice_stripe.test_clock.cleanup` task backstops by deleting
    tagged test clocks older than a configurable threshold.

    ## Metadata marker

    Every clock created via `test_clock/1` is tagged with the metadata
    key-value `"lattice_stripe_test_clock" => "v1"` (stored as the module
    attribute `@cleanup_marker`). The Mix task uses this marker to identify
    LatticeStripe-managed clocks vs user-created ones. Changing the marker
    key or value is a **breaking change**.

    Stripe limits metadata to 50 keys per object. If user-supplied metadata
    + the marker would exceed 50, `test_clock/1` raises a
    `LatticeStripe.Testing.TestClockError` rather than silently dropping
    the marker.

    ## Supported advance units (v1)

    `advance/2` accepts: `:seconds`, `:minutes`, `:hours`, `:days`, or `:to`
    (absolute `DateTime`). Passing `:months` or `:years` raises
    `ArgumentError` — Elixir 1.15 (the project minimum) lacks `Date.shift/2`,
    and month-length arithmetic is fiddly. For month/year advancement, use:

        advance(clock, to: DateTime.utc_now() |> DateTime.add(86_400 * 30, :second))

    ## Client override

    Every helper accepts a per-call `client:` option that wins over the
    compile-time binding. This supports multi-account test scenarios:

        test_clock(frozen_time: ts, client: other_client)
    """

    alias LatticeStripe.TestHelpers.TestClock, as: Backend
    alias LatticeStripe.Testing.TestClock.{Owner, Error}

    @cleanup_marker {"lattice_stripe_test_clock", "v1"}
    @max_metadata_keys 50

    @doc false
    def cleanup_marker, do: @cleanup_marker

    # ---------------------------------------------------------------------
    # __using__ macro
    # ---------------------------------------------------------------------

    defmacro __using__(opts) do
      client = Keyword.fetch!(opts, :client)

      unless is_atom(client) do
        raise CompileError,
          description:
            "LatticeStripe.Testing.TestClock requires :client to be a module atom (the caller that resolves MyApp.StripeClient to a %Client{}), got: #{inspect(client)}"
      end

      quote do
        import LatticeStripe.Testing.TestClock,
          only: [
            test_clock: 0,
            test_clock: 1,
            advance: 2,
            freeze: 1,
            create_customer: 2,
            create_customer: 3,
            with_test_clock: 1
          ]

        # Store the compile-time bound client on the test module as a module attr.
        # Helper functions resolve it lazily via apply/2.
        @__lattice_test_clock_client__ unquote(client)

        def __lattice_test_clock_client__, do: @__lattice_test_clock_client__
      end
    end

    # ---------------------------------------------------------------------
    # Client resolution
    # ---------------------------------------------------------------------

    # Resolve a client: per-call override > compile-time bound > error.
    # The bound client is stored on the calling test module as an attr;
    # since helper functions are imported, the caller IS the test module —
    # use Process.put/Process.get to stash the resolved client per test.
    #
    # ACTUAL RESOLUTION PATTERN: the use-macro injects a module function
    # `__lattice_test_clock_client__/0` on the test module. Helpers look up
    # the calling test module via `self()` process dict... NO, that doesn't
    # work under async. The cleanest pattern is to have the use-macro
    # generate wrapper functions in the test module that CAPTURE the
    # compile-time client and call through to private helpers here.
    #
    # STUDY Oban.Testing's github source for the exact shape. For v1,
    # implement the following pragmatic approach:
    #
    # - The use-macro imports `test_clock/0,1` etc. AND injects
    #   `__lattice_test_clock_client__/0` returning the bound module.
    # - `test_clock/1` resolves the client by looking at the process dict
    #   key `:__lattice_stripe_bound_client__` (set by `with_test_clock/1`
    #   during setup) OR by the per-call override opt. The use-macro is
    #   responsible for wiring `with_test_clock/1` to set the process dict
    #   before returning context.
    #
    # ALTERNATIVE (cleaner, recommended): the use-macro generates
    # wrapper functions directly in the test module that capture the
    # client. Example:
    #
    #   quote do
    #     def test_clock(opts \\ []) do
    #       LatticeStripe.Testing.TestClock.__test_clock__(
    #         opts,

    ## Options

    - `:frozen_time` — unix timestamp integer (default: current system time)
    - `:name` — human-readable name (default: "lattice_stripe_test")
    - `:metadata` — map of custom metadata (merged with the cleanup marker)
    - `:client` — per-call client override

    defp compute_frozen_time!(clock, unit_opts) do
      cond do
        Keyword.has_key?(unit_opts, :to) ->
          case Keyword.fetch!(unit_opts, :to) do
            %DateTime{} = dt -> DateTime.to_unix(dt)
            other -> raise ArgumentError, ":to must be a DateTime, got #{inspect(other)}"
          end

        Keyword.has_key?(unit_opts, :months) or Keyword.has_key?(unit_opts, :years) ->
          raise ArgumentError,
                ":months and :years are not supported in v1 (Elixir 1.15 lacks Date.shift/2). Use `to: DateTime.utc_now() |> DateTime.add(86_400 * 30, :second)` or similar."

        true ->
          delta =
            cond do
              n = Keyword.get(unit_opts, :seconds) -> n
              n = Keyword.get(unit_opts, :minutes) -> n * 60
              n = Keyword.get(unit_opts, :hours) -> n * 3_600
              n = Keyword.get(unit_opts, :days) -> n * 86_400
              true ->
                raise ArgumentError,
                      "advance/2 unit_opts must contain one of :seconds, :minutes, :hours, :days, :to — got #{inspect(unit_opts)}"
            end

          (clock.frozen_time || System.system_time(:second)) + delta
      end
    end

    defp merge_cleanup_marker(user_metadata) when is_map(user_metadata) do
      {marker_key, marker_val} = @cleanup_marker

      merged = Map.put(user_metadata, marker_key, marker_val)

      if map_size(merged) > @max_metadata_keys do
        raise Error,
          message:
            "Merging the LatticeStripe cleanup marker into metadata would exceed Stripe's 50-key limit. Reduce user-supplied metadata to 49 keys or fewer.",
          type: :metadata_limit
      end

      merged
    end

    defp resolve_client!(opts) do
      case Keyword.get(opts, :client) do
        nil ->
          # Look up the compile-time bound client. This requires the
          # use-macro to have injected a process-dict entry or a wrapper
          # function — see the __using__ alternative pattern above.
          # Pragmatic v1: read from the process dict key the use-macro sets.
          case Process.get(:__lattice_stripe_bound_client__) do
            nil ->
              raise Error,
                message:
                  "No LatticeStripe client is bound. Either `use LatticeStripe.Testing.TestClock, client: MyApp.StripeClient` in your CaseTemplate, or pass `client:` per call.",
                type: :no_client_bound

            mod when is_atom(mod) ->
              apply(mod, :stripe_client, [])
          end

        %LatticeStripe.Client{} = client ->
          client

        mod when is_atom(mod) ->
          apply(mod, :stripe_client, [])
      end
    end

    defp ensure_owner do
      case Process.get(:__lattice_stripe_test_clock_owner__) do
        nil ->
          owner = Owner.start_owner!()
          Process.put(:__lattice_stripe_test_clock_owner__, owner)

          ExUnit.Callbacks.on_exit(fn ->
            client = Process.get(:__lattice_stripe_bound_client__) |> resolve_client_for_cleanup()
            if client, do: Owner.cleanup(owner, client)
          end)

          owner

        owner when is_pid(owner) ->
          owner
      end
    end

    defp resolve_client_for_cleanup(nil), do: nil
    defp resolve_client_for_cleanup(mod) when is_atom(mod), do: apply(mod, :stripe_client, [])
  end
  ```

  ```elixir
  @default_older_than_ms 3_600_000

  @doc """
  Lists and optionally deletes LatticeStripe-tagged test clocks.

  This is the shared deletion core used by both
  `LatticeStripe.Testing.TestClock.Owner.cleanup/2` and
  `mix lattice_stripe.test_clock.cleanup`. See those callers for the
  user-facing entry points.

  ## Options

  - `:older_than_ms` — only consider clocks older than N milliseconds
    (default: 3_600_000 = 1 hour)
  - `:delete` — `true` to actually delete, `false` to return candidates
    only (default: false)
  - `:marker` — override the cleanup marker tuple (default: all versions
    of the LatticeStripe marker: v1, v2)

  ## Returns

  - `:delete => false` — `{:ok, [%TestClock{}, ...]}` (candidates only)
  - `:delete => true` — `{:ok, %{deleted: n, failed: n}}`
  """
  @spec cleanup_tagged(Client.t(), keyword()) :: {:ok, term()} | {:error, Error.t()}
  def cleanup_tagged(%Client{} = client, opts \\ []) do
    older_than_ms = Keyword.get(opts, :older_than_ms, @default_older_than_ms)
    delete? = Keyword.get(opts, :delete, false)
    now = System.system_time(:second)
    cutoff = now - div(older_than_ms, 1000)

    # Stream over ALL clocks for this account. stream! takes care of pagination.
    matching =
      client
      |> stream!(%{}, [])
      |> Enum.filter(fn clock ->
        has_marker?(clock) and old_enough?(clock, cutoff)
      end)

    if delete? do
      {deleted, failed} = do_delete_each(client, matching)
      {:ok, %{deleted: deleted, failed: failed, total_matched: length(matching)}}
    else
      {:ok, matching}
    end
  end

  defp has_marker?(%__MODULE__{metadata: nil}), do: false

  defp has_marker?(%__MODULE__{metadata: meta}) when is_map(meta) do
    # Accept v1 and v2 (and any future versions) per D-13g.
    Map.get(meta, "lattice_stripe_test_clock") in ["v1", "v2"]
  end

  defp has_marker?(_), do: false

  defp old_enough?(%__MODULE__{created: created}, cutoff) when is_integer(created) do
    created <= cutoff
  end

  defp old_enough?(_, _), do: false

  defp do_delete_each(client, clocks) do
    Enum.reduce(clocks, {0, 0}, fn clock, {d, f} ->
      case delete(client, clock.id) do
        {:ok, _} -> {d + 1, f}
        {:error, _} -> {d, f + 1}
      end
    end)
  end
  ```

  ```elixir
    scenarios that bypass the ExUnit `on_exit` per-test cleanup. Stripe
    enforces a hard limit of 100 test clocks per account — if leaked
    clocks pile up, every subsequent test fails at create time.

        mix lattice_stripe.test_clock.cleanup --client MyApp.StripeClient --older-than 2h

    - `--client` — module name (e.g. `MyApp.StripeClient`) or Application
      env key exposing a `stripe_client/0` function returning a `%Client{}`
    - `--dry-run` — `true` (default) means list candidates without deleting
    - `--older-than` — duration with unit (`30m`, `1h`, `2h`, `24h`). Default: `1h`
    - `--yes` — explicit confirmation required for destructive delete; default `false`

    ## Safety

    Destructive delete requires BOTH `--yes` and `--dry-run false`. Without
    `--yes` the task always exits after printing candidates. Without
    `--dry-run false` the task exits after printing. This is intentional —
    see the threat model in `.planning/phases/13-billing-test-clocks`.

    If the client's base_url points at stripe-mock (localhost:12111 or
    similar), the task prints a no-op message and exits cleanly.

    @impl Mix.Task

      {opts, _, _} =
        OptionParser.parse(args,
          strict: [client: :string, dry_run: :boolean, older_than: :string, yes: :boolean],
          aliases: [c: :client, n: :dry_run, y: :yes]
        )

      Mix.Task.run("app.start")

      client_spec = Keyword.get(opts, :client) || Mix.raise("--client is required")
      dry_run = Keyword.get(opts, :dry_run, true)

        Mix.shell().info("Detected stripe-mock at #{client.base_url}; no-op cleanup.")
        :ok
      else
        do_cleanup(client, dry_run, yes, older_than_ms)
      end
    end

             older_than_ms: older_than_ms,
             delete: false

            true ->
              {:ok, %{deleted: d, failed: f}} =
                LatticeStripe.TestHelpers.TestClock.cleanup_tagged(client,
                  older_than_ms: older_than_ms,
                  delete: true
                )

              Mix.shell().info("Deleted #{d} clock(s); #{f} failure(s).")
          end

        {:error, err} ->
          Mix.raise("Cleanup failed: #{inspect(err)}")
      end
    end

        "Elixir." <> _ = mod_string ->
          apply(String.to_existing_atom(mod_string), :stripe_client, [])

        mod_string ->
          mod = String.to_existing_atom("Elixir." <> mod_string)
          apply(mod, :stripe_client, [])
      end
    rescue
      e in [ArgumentError, UndefinedFunctionError] ->
        Mix.raise(
          "Could not resolve --client #{inspect(spec)}: expected a module name exposing stripe_client/0. (#{Exception.message(e)})"
        )
    end

    defp stripe_mock?(%LatticeStripe.Client{base_url: url}) when is_binary(url) do
      url =~ "localhost" or url =~ "127.0.0.1" or url =~ "stripe-mock"
    end

    defp stripe_mock?(_), do: false

    defp parse_duration!(str) do
      case Regex.run(~r/^(\d+)([smhd])$/, str) do
        [_, n, "s"] -> String.to_integer(n) * 1_000
        [_, n, "m"] -> String.to_integer(n) * 60_000
        [_, n, "h"] -> String.to_integer(n) * 3_600_000
        [_, n, "d"] -> String.to_integer(n) * 86_400_000
        _ -> Mix.raise("Invalid --older-than: #{str}. Use N[smhd], e.g. 30m, 1h, 24h, 7d.")
      end
    end
  end
  ```
