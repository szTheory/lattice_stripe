      plug Plug.Parsers,

    Implementation:
    ```elixir
    def read_body(conn, opts) do
      case Plug.Conn.read_body(conn, opts) do
        {:ok, body, conn} ->
          conn = Plug.Conn.put_private(conn, :raw_body, body)
          {:ok, body, conn}

        {:more, body, conn} ->
          conn = Plug.Conn.put_private(conn, :raw_body, body)
          {:more, body, conn}

        {:error, reason} ->
          {:error, reason}
      end
    end
    ```

    **`init/1` (D-14, D-15):**
    ```elixir
    @impl Plug
    def init(opts) do
      validated = NimbleOptions.validate!(opts, @schema)
      path_info =
        case Keyword.get(validated, :at) do
          nil -> nil
          at -> String.split(at, "/", trim: true)
        end
      validated |> Map.new() |> Map.put(:path_info, path_info)
    end
    ```

    **`call/2` — Four clauses in order (D-12, D-13, D-15):**

    1. **POST + matching path (or no path filter):**
       ```elixir
       # When `at:` is set and path matches
       def call(%Plug.Conn{method: "POST", path_info: path_info} = conn, %{path_info: path_info} = opts)
           when path_info != nil do
         handle_webhook(conn, opts)
       end

       # When no `at:` is set (process all POSTs)
       def call(%Plug.Conn{method: "POST"} = conn, %{path_info: nil} = opts) do
         handle_webhook(conn, opts)
       end
       ```

    2. **Non-POST + matching path -> 405 (D-13):**
       ```elixir
       def call(%Plug.Conn{path_info: path_info} = conn, %{path_info: path_info})
           when path_info != nil do
         conn
         |> Plug.Conn.put_resp_header("allow", "POST")
         |> Plug.Conn.send_resp(405, "Method Not Allowed")
         |> Plug.Conn.halt()
       end
       ```

    3. **Non-POST + no path filter -> pass through:**
       ```elixir
       def call(conn, %{path_info: nil}), do: conn
       ```

    4. **Path doesn't match -> pass through:**
       ```elixir
       def call(conn, _opts), do: conn
       ```

    **`defp handle_webhook/2`:**
    - Resolve secret: `resolve_secret(opts.secret)` (D-19)
    - Read raw body: `get_raw_body(conn)` — check `conn.private[:raw_body]` first, fallback to `Plug.Conn.read_body(conn)` (D-17)
    - Get sig header: `Plug.Conn.get_req_header(conn, "stripe-signature") |> List.first()`
    - Call `LatticeStripe.Webhook.construct_event(raw_body, sig_header, secret, tolerance: opts.tolerance)`
    - On `{:ok, event}`:
      - Assign: `conn = Plug.Conn.assign(conn, :stripe_event, event)` (D-16)
      - If `opts.handler` is nil: return conn (pass-through mode, D-10)
      - If handler set: call `opts.handler.handle_event(event)` and handle return (D-12):
        - `:ok` or `{:ok, _}` -> `send_resp(conn, 200, "") |> halt()`
        - `:error` or `{:error, _}` -> `send_resp(conn, 400, "") |> halt()`
        - Other -> `raise RuntimeError, "Expected handle_event/1 to return :ok | {:ok, term} | :error | {:error, term}, got: #{inspect(result)}"`
    - On `{:error, _reason}`:
      - `send_resp(conn, 400, "") |> halt()` (D-12)

    **`defp resolve_secret/1` (D-19):**
    ```elixir
    defp resolve_secret({mod, fun, args}), do: apply(mod, fun, args)
    defp resolve_secret(fun) when is_function(fun, 0), do: fun.()
    defp resolve_secret(secret), do: secret
    ```

    **`defp get_raw_body/1` (D-17):**
    ```elixir
    defp get_raw_body(conn) do
      case conn.private[:raw_body] do
        nil ->
          case Plug.Conn.read_body(conn) do
            {:ok, body, _conn} -> body
            _ -> ""
          end
        body -> body
      end
    end
    ```
