Add a private helper that checks params and emits telemetry before the HTTP call:

```elixir
def create(%Client{} = client, params \\ %{}, opts \\ []) do
  maybe_emit_auto_advance_telemetry(client, params)

  %Request{method: :post, path: "/v1/invoices", params: params, opts: opts}
  |> then(&Client.request(client, &1))
  |> Resource.unwrap_singular(&from_map/1)
end

defp maybe_emit_auto_advance_telemetry(%Client{telemetry_enabled: false}, _params), do: :ok
defp maybe_emit_auto_advance_telemetry(%Client{}, params) do
  unless Map.has_key?(params, "auto_advance") do
    :telemetry.execute(
      [:lattice_stripe, :invoice, :auto_advance_defaulted],
      %{system_time: System.system_time()},
      %{resource: "invoice", operation: "create", auto_advance: :defaulted}
    )
  end
end
```

Per D-14c: this fires BEFORE the HTTP call. Checks `Map.has_key?(params, "auto_advance")` — presence of the key is what matters, not the value.
