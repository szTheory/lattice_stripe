    **4. Retry loop (D-14, D-15, D-16):**
    ```elixir
    defp do_request_with_retries(client, transport_request, method, idempotency_key, max_retries) do
      do_request_with_retries(client, transport_request, method, idempotency_key, max_retries, _attempt = 1, _total_attempts = 1)
    end

    defp do_request_with_retries(client, transport_request, method, idempotency_key, max_retries, attempt, total_attempts) do
      case do_request(client, transport_request) do
        {:ok, _} = success ->
          success

        {:error, %Error{} = error} = failure ->
          if attempt <= max_retries do
            context = %{
              error: error,
              status: error.status,
              headers: get_last_response_headers(error),
              stripe_should_retry: nil,  # parsed from headers
              method: method,
              idempotency_key: idempotency_key
            }
            # Parse Stripe-Should-Retry from response headers stored in context
            context = put_stripe_should_retry(context)

            case client.retry_strategy.retry?(attempt, context) do
              {:retry, delay_ms} ->
                emit_retry_telemetry(client, method, transport_request, error, attempt, delay_ms)
                Process.sleep(delay_ms)
                do_request_with_retries(client, transport_request, method, idempotency_key, max_retries, attempt + 1, total_attempts + 1)

              :stop ->
                failure
            end
          else
            failure
          end
      end
    end
    ```

    **Important implementation detail for response headers in retry context:**

    - Option A: Store response headers on the Error struct (add `:response_headers` field — but this leaks internal details)
    - Option B: Return a 3-tuple `{:error, error, headers}` from `do_request` internally, unwrap to `{:error, error}` at the public boundary

    Use Option B — keep response headers as an internal detail of the retry loop. Modify `do_request/2` to return `{:ok, decoded}` or `{:error, error, resp_headers}`:
